import { Tournament, Team, Match, TournamentFormat, MatchStatus, TeamSize } from '../types';
 
const STORAGE_KEY = 'rivals_arena_data';
 
// --- Helper: Load/Save ---
const loadData = (): Tournament[] => {
  try {
    const existing = localStorage.getItem(STORAGE_KEY);
    return existing ? JSON.parse(existing) : [];
  } catch (e) {
    console.error("Failed to load tournament data", e);
    return [];
  }
};
 
const saveData = (data: Tournament[]) => {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(data));
};
 
// --- Helper: Seeding Algorithm ---
// Returns an array of team indices in standard seeded order (1v8, 4v5, 2v7, 3v6 etc.)
const getSeededOrders = (numTeams: number): number[] => {
  let rounds = Math.log2(numTeams);
  let order = [1, 2]; // Initial seeds for 2 teams
 
  for (let i = 0; i < rounds - 1; i++) {
    let nextOrder: number[] = [];
    const sum = order.length * 2 + 1;
    for (let j = 0; j < order.length; j++) {
      nextOrder.push(order[j]);
      nextOrder.push(sum - order[j]);
    }
    order = nextOrder;
  }
  // Convert 1-based seeds to 0-based indices
  return order.map(n => n - 1);
};
 
// --- Helper: Bracket Logic ---
const generateSingleEliminationBracket = (teams: Team[]): Match[][] => {
  if (teams.length < 2) return [];
 
  // 1. Sort teams by skill level (Descending)
  const sortedTeams = [...teams].sort((a, b) => (b.skillLevel || 0) - (a.skillLevel || 0));
 
  // 2. Calculate bracket size (next power of 2)
  const teamCount = teams.length;
  const bracketSize = Math.pow(2, Math.ceil(Math.log2(teamCount)));
  const roundsCount = Math.log2(bracketSize);
  
  // 3. Get seeding order (e.g. for 8 slots: indices 0, 7, 3, 4, 1, 6, 2, 5)
  // Indices refer to the sortedTeams array.
  const seedIndices = getSeededOrders(bracketSize);
 
  // 4. Initialize Matches Structure
  const matches: Match[][] = [];
  
  for (let r = 0; r < roundsCount; r++) {
    const matchesInRound = bracketSize / Math.pow(2, r + 1);
    const roundMatches: Match[] = [];
    for (let m = 0; m < matchesInRound; m++) {
      roundMatches.push({
        id: `m-${Date.now()}-${r}-${m}`,
        roundIndex: r,
        matchIndex: m,
        teamA: null,
        teamB: null,
        scoreA: 0,
        scoreB: 0,
        status: MatchStatus.PENDING,
      });
    }
    matches.push(roundMatches);
  }
 
  // 5. Link Matches (Winner advancement)
  for (let r = 0; r < roundsCount - 1; r++) {
    for (let m = 0; m < matches[r].length; m++) {
      const currentMatch = matches[r][m];
      const nextRoundIndex = r + 1;
      const nextMatchIndex = Math.floor(m / 2);
      const position = m % 2 === 0 ? 'A' : 'B';
      
      const nextMatch = matches[nextRoundIndex][nextMatchIndex];
      currentMatch.nextMatchId = nextMatch.id;
      currentMatch.nextMatchPosition = position;
    }
  }
 
  // 6. Populate Round 1 with Seeded Teams
  // We pair seeds based on the standard order: (Seed 1 vs Seed N), (Seed X vs Seed Y)
  // For the visual layout "Left vs Right" where final is in middle:
  // The seeding order function above returns pairs sequentially (Match 1, Match 2...)
  // We need to ensure that Seed 1 is in the first match (Top Left) and Seed 2 is in the middle match (Top Right of other bracket half)
  
  // However, standard seeding logic (1v8, 4v5, 3v6, 2v7) puts 1 and 2 at opposite ends of the array.
  // Match 0: 1 vs 8
  // Match 1: 4 vs 5
  // Match 2: 3 vs 6
  // Match 3: 2 vs 7
  // If we render Matches 0-1 on Left and Matches 2-3 on Right, Seed 1 is Left, Seed 2 is Right. Ideally they meet in final.
  // Logic checks out.
 
  const round1 = matches[0];
  
  // Iterate through matches and fill slots
  for (let i = 0; i < round1.length; i++) {
    // Each match takes 2 seeds from our seedIndices list
    // e.g. Match 0 takes seedIndices[0] and seedIndices[1] (which are Seed 1 and Seed 8)
    const seedIndexA = seedIndices[i * 2];
    const seedIndexB = seedIndices[i * 2 + 1];
 
    // Assign Team A
    if (seedIndexA < sortedTeams.length) {
      round1[i].teamA = sortedTeams[seedIndexA];
      round1[i].teamA!.seed = seedIndexA + 1;
    }
 
    // Assign Team B
    if (seedIndexB < sortedTeams.length) {
      round1[i].teamB = sortedTeams[seedIndexB];
      round1[i].teamB!.seed = seedIndexB + 1;
    } else {
      // BYE (No opponent)
      round1[i].status = MatchStatus.BYE;
      round1[i].winnerId = round1[i].teamA?.id;
      
      // Auto-advance
      const nextMatch = matches[1]?.find(nm => nm.id === round1[i].nextMatchId);
      if (nextMatch && round1[i].teamA) {
        if (round1[i].nextMatchPosition === 'A') nextMatch.teamA = round1[i].teamA;
        else nextMatch.teamB = round1[i].teamA;
      }
    }
  }
 
  return matches;
};
 
// --- Service Methods ---
 
export const tournamentService = {
  getAll: (): Tournament[] => {
    return loadData();
  },
 
  getById: (id: string): Tournament | undefined => {
    return loadData().find(t => t.id === id);
  },
 
  create: (data: Partial<Tournament>): Tournament => {
    const tournaments = loadData();
    const newTournament: Tournament = {
      id: `t-${Date.now()}`,
      name: data.name?.trim() || 'Untitled Tournament',
      description: data.description || '',
      game: 'Roblox Rivals',
      format: data.format || TournamentFormat.SINGLE_ELIMINATION,
      teamSize: data.teamSize || TeamSize.SOLO,
      maxTeams: data.maxTeams || 16,
      status: 'Open',
      startDate: data.startDate || new Date().toISOString(),
      prizePool: data.prizePool || 'None',
      teams: [],
      matches: []
    };
    tournaments.push(newTournament);
    saveData(tournaments);
    return newTournament;
  },
 
  delete: (id: string): void => {
    const tournaments = loadData();
    const filtered = tournaments.filter(t => t.id !== id);
    saveData(filtered);
  },
 
  startTournament: (id: string): { success: boolean, message: string } => {
    const tournaments = loadData();
    const tIdx = tournaments.findIndex(t => t.id === id);
    if (tIdx === -1) return { success: false, message: 'Tournament not found' };
    
    const tournament = tournaments[tIdx];
    
    if (tournament.teams.length < 2) {
      return { success: false, message: 'Need at least 2 teams to start.' };
    }
 
    tournament.status = 'Active';
    tournament.matches = generateSingleEliminationBracket(tournament.teams);
    
    tournaments[tIdx] = tournament;
    saveData(tournaments);
    return { success: true, message: 'Tournament started!' };
  },
 
  addTeam: (tournamentId: string, teamName: string, username: string, level: number): { success: boolean, message: string } => {
    const tournaments = loadData();
    const idx = tournaments.findIndex(t => t.id === tournamentId);
    if (idx === -1) return { success: false, message: 'Tournament not found' };
 
    const t = tournaments[idx];
    if (t.status !== 'Open') return { success: false, message: 'Registrations are closed' };
    if (t.teams.length >= t.maxTeams) return { success: false, message: 'Tournament is full' };
 
    // Check duplicate name
    if (t.teams.some(team => team.name.toLowerCase() === teamName.toLowerCase())) {
       return { success: false, message: 'Team name already taken' };
    }
 
    const newTeam: Team = {
      id: `team-${Date.now()}`,
      name: teamName,
      skillLevel: level,
      players: [{ id: `p-${Date.now()}`, username }]
    };
 
    t.teams.push(newTeam);
    tournaments[idx] = t;
    saveData(tournaments);
    return { success: true, message: 'Team registered successfully' };
  },
 
  removeTeam: (tournamentId: string, teamId: string): void => {
    const tournaments = loadData();
    const tIdx = tournaments.findIndex(t => t.id === tournamentId);
    if (tIdx === -1) return;
    
    if (tournaments[tIdx].status !== 'Open') return; // Can't remove if active
 
    tournaments[tIdx].teams = tournaments[tIdx].teams.filter(t => t.id !== teamId);
    saveData(tournaments);
  },
 
  updateMatch: (tournamentId: string, matchId: string, scoreA: number, scoreB: number, winnerId: string): Tournament | null => {
    const tournaments = loadData();
    const tIdx = tournaments.findIndex(t => t.id === tournamentId);
    if (tIdx === -1) return null;
 
    const tournament = tournaments[tIdx];
    
    // Find match in any round
    let match: Match | undefined;
    let rIdx = -1;
    
    for (let r = 0; r < tournament.matches.length; r++) {
       const m = tournament.matches[r].find(m => m.id === matchId);
       if (m) {
         match = m;
         rIdx = r;
         break;
       }
    }
 
    if (!match) return null;
 
    // Logic Update
    match.scoreA = scoreA;
    match.scoreB = scoreB;
    match.winnerId = winnerId;
    match.status = MatchStatus.COMPLETED;
 
    // Advance Logic
    if (match.nextMatchId) {
      // Find next match
      for (let r = rIdx + 1; r < tournament.matches.length; r++) {
        const nextMatch = tournament.matches[r].find(nm => nm.id === match.nextMatchId);
        if (nextMatch) {
          const winningTeam = match.teamA?.id === winnerId ? match.teamA : match.teamB;
          
          if (match.nextMatchPosition === 'A') {
            nextMatch.teamA = winningTeam;
          } else {
            nextMatch.teamB = winningTeam;
          }
          
          // Reset status if it was completed (re-play scenario) or set to pending
          if (nextMatch.status === MatchStatus.COMPLETED) {
             nextMatch.status = MatchStatus.PENDING; 
             nextMatch.winnerId = undefined;
             nextMatch.scoreA = 0;
             nextMatch.scoreB = 0;
          }
          break;
        }
      }
    } else {
      // Grand Final Finished
      tournament.winnerTeamId = winnerId;
      tournament.status = 'Completed';
    }
 
    tournaments[tIdx] = tournament;
    saveData(tournaments);
    return tournament;
  }
};
